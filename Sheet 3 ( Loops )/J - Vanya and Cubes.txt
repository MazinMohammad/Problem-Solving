#include <iostream>
#include <cmath>
using namespace std;

int main()
{
   int n, lvl = 1;
   cin >> n;

   while (n>0)
   {
      int cube = ((lvl*(lvl+1))/2);
      if (n>=cube)
         n-=cube;
      lvl++;
   }  
   cout << lvl-1;
}
